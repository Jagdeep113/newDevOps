# Set user information
git config --global user.name "Your Name"
git config --global user.email "your@email.com"

# View configuration
git config --list
git config user.name  # View specific setting

# Set default editor
git config --global core.editor "code --wait"

# Set default branch name
git config --global init.defaultBranch main

# Create aliases
git config --global alias.co checkout
git config --global alias.br branch
git config --global alias.ci commit
git config --global alias.st status

Repository Management

# Initialize new repository
git init

# Clone a repository
git clone <repository-url>
git clone <repository-url> <directory-name>
git clone --depth 1 <url>  # Shallow clone (only latest commit)

# Check repository status
git status
git status -s  # Short format
git status -u  # Show untracked files

# Remove repository metadata (danger!)
rm -rf .git

Basic Workflow

# Add files to staging area
git add <file>
git add .  # Add all files
git add *.js  # Add all JS files
git add -A  # Add all (including deletions)
git add -u  # Add only modified/deleted files

# Commit changes
git commit -m "Commit message"
git commit -am "Add and commit in one step"  # Only for tracked files
git commit --amend  # Modify last commit
git commit --amend --no-edit  # Amend without changing message

# View commits
git log
git log --oneline  # Compact view
git log --graph --oneline --all  # Visual history
git log -p  # Show changes
git log --stat  # Show stats
git log --since="2 weeks ago"
git log --author="name"

Branching & Merging

# List branches
git branch
git branch -a  # Show all branches (including remote)
git branch -v  # Show last commit on each branch
git branch --merged  # Show merged branches

# Create branches
git branch <branch-name>
git checkout -b <branch-name>  # Create and switch
git switch -c <branch-name>   # Alternative (Git 2.23+)

# Switch branches
git checkout <branch-name>
git switch <branch-name>  # Alternative (Git 2.23+)

# Delete branches
git branch -d <branch-name>  # Safe delete (merged only)
git branch -D <branch-name>  # Force delete
git push origin --delete <branch-name>  # Delete remote branch

# Rename branch
git branch -m <new-name>  # Rename current branch
git branch -m <old-name> <new-name>

# Merge branches
git merge <branch-name>
git merge --no-ff <branch-name>  # Always create merge commit
git merge --squash <branch-name>  # Squash all commits into one

# Rebase (rewrite history)
git rebase <branch-name>
git rebase -i HEAD~3  # Interactive rebase last 3 commits

# Stash changes
git stash  # Save changes temporarily
git stash save "message"
git stash list
git stash pop  # Apply and remove
git stash apply  # Apply but keep in stash
git stash drop  # Remove specific stash
git stash clear  # Remove all stashes

Remote Repositories

# Manage remotes
git remote add origin <url>
git remote -v  # View remotes
git remote show origin  # Show remote info
git remote rename <old> <new>
git remote remove <name>

# Fetch changes
git fetch  # Fetch all remotes
git fetch origin  # Fetch from specific remote
git fetch --prune  # Remove deleted remote branches

# Pull changes
git pull  # Fetch + merge
git pull origin <branch>
git pull --rebase  # Pull with rebase instead of merge

# Push changes
git push
git push origin <branch>
git push -u origin <branch>  # Set upstream branch
git push --force  # Force push (danger!)
git push --force-with-lease  # Safer force push

# Track remote branches
git branch -u origin/<branch>  # Set upstream
git branch -vv  # View tracking branches

Comparing & Inspecting

# Compare changes
git diff  # Unstaged changes
git diff --staged  # Staged changes
git diff HEAD  # All changes since last commit
git diff <commit1> <commit2>
git diff <branch1>..<branch2>
git diff --name-only  # Show only filenames

# Show specific commit
git show <commit>
git show HEAD~1  # Show previous commit
git show <commit>:<file>  # Show file at specific commit

# Blame (who changed what)
git blame <file>
git blame -L 10,20 <file>  # Specific lines


 Undoing Changes

 # Undo unstaged changes
git checkout -- <file>  # Discard changes
git restore <file>  # Alternative (Git 2.23+)

# Unstage files
git reset HEAD <file>
git restore --staged <file>  # Alternative

# Reset (rewind commits)
git reset --soft HEAD~1  # Keep changes staged
git reset --mixed HEAD~1  # Keep changes unstaged (default)
git reset --hard HEAD~1  # Discard all changes (danger!)

# Revert (safe undo)
git revert <commit>  # Create new commit undoing changes
git revert --no-commit <commit>  # Revert without committing

# Clean untracked files
git clean -n  # Show what would be deleted
git clean -f  # Delete untracked files
git clean -fd  # Delete untracked files and directories

Tagging

# Create tags
git tag v1.0.0  # Lightweight tag
git tag -a v1.0.0 -m "Version 1.0.0"  # Annotated tag
git tag -a v1.0.0 <commit>  # Tag specific commit

# List tags
git tag
git tag -l "v1.*"  # Filter tags

# Show tag info
git show v1.0.0

# Delete tags
git tag -d v1.0.0  # Local delete
git push origin --delete v1.0.0  # Remote delete

# Push tags
git push origin v1.0.0  # Single tag
git push origin --tags  # All tags
